	fire_only_once = yes
	trigger = {
		tag = BAK
		has_civil_war = yes
		is_subject = no
		has_government = communism
	}
	mean_time_to_happen = { days = 30 }
	option = {
		name = ww1_azerbaijan.2.a
		if = {
			limit = {
				RUS = {

				}
			}

		}	
		set_politics = {
			ruling_party = neutrality
			elections_allowed = no
		}	
		add_popularity = {
			ideology = neutrality
			popularity = 0.2
		}
	}
}
