# Collapse of Baltic Duchy
#############################
country_event = {
	id = ww1_estonia.1
	title = ww1_estonia.1.t
	desc = ww1_estonia.1.d
	picture = GFX_report_event_ger_civil_war
	trigger = {
		tag = GER
		has_country_flag = great_war_loser
		OR = {
			AND = {
				owns_state = 13
				owns_state = 191
				owns_state = 12
			}
			UBD = {
				exists = yes
			}
		}
	}
	fire_only_once = yes
	mean_time_to_happen = {
		days = 31
	}
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					LAT = {
						OR = {
							has_country_flag = LAT_iskolata
							has_cosmetic_tag = LAT_iskolata
						}
						exists = no
					}
				}
				LAT = {
					clr_country_flag = LAT_iskolata
					drop_cosmetic_tag = yes
					promote_character = LAT_peteris_stucka
					set_politics = {
						ruling_party = democratic
						elections_allowed = yes
					}
					set_rule = {
						can_join_factions = yes
					}
					set_popularities = {
						democratic = 50
						neutrality = 30
						fascism = 0
						communism = 20
					}
				}
			}
			every_state = {
				limit = {
					OR = {
						is_core_of = LIT
						is_core_of = LAT
						is_core_of = EST
					}
					NOT = {
						state = 96
					}
				}
				remove_russian_cores = yes
				remove_russian_claims = yes
			}
			every_state = {
				limit = {
					OR = {
						is_owned_by = GER
						is_owned_by = UBD
						is_owned_by = COU
						is_owned_by = SOV
						is_owned_by = RUS
					}
				}
				if = {
					limit = {
						OR = {
							state = 13
							state = 191
						}
					}
					transfer_state_to = EST
				}
				if = {
					limit = {
						OR = {
							state = 12
							#state = 96 #not now
							state = 190
						}
					}
					transfer_state_to = LAT
				}
			}
			set_global_flag = estonia_released
			EST = {
				division_template = {
					name = "Jalaväediviisi"				# Infantry Division	
					division_names_group = EST_INF_01
					# Note: Divisions were nowhere near full strength until mobilized
					regiments = {
						infantry = { x = 0 y = 0 }
						infantry = { x = 0 y = 1 }
						infantry = { x = 0 y = 2 }
						infantry = { x = 1 y = 0 }
						infantry = { x = 1 y = 1 }
						infantry = { x = 1 y = 2 }
					}
				}
				capital_scope = {
					create_unit = {
						division = "name = \"I. Jalaväediviisi\" division_template = \"Jalaväediviisi\" start_experience_factor = 1.0 start_equipment_factor = 1.0"
						owner = PREV
					}
					create_unit = {
						division = "name = \"II. Jalaväediviisi\" division_template = \"Jalaväediviisi\" start_experience_factor = 1.0 start_equipment_factor = 1.0"
						owner = PREV
					}
					create_unit = {
						division = "name = \"III. Jalaväediviisi\" division_template = \"Jalaväediviisi\" start_experience_factor = 1.0 start_equipment_factor = 1.0"
						owner = PREV
					}
				}
			}
			set_global_flag = latvia_released
			LAT = {
				division_template = {
					name = "Kajnieku Divizija"				# Infantry Division	
					division_names_group = LAT_INF_01
					# Note: Divisions were nowhere near full strength until mobilized
					regiments = {
						infantry = { x = 0 y = 0 }
						infantry = { x = 0 y = 1 }
						infantry = { x = 0 y = 2 }
						infantry = { x = 1 y = 0 }
						infantry = { x = 1 y = 1 }
						infantry = { x = 1 y = 2 }
						infantry = { x = 2 y = 0 }
						infantry = { x = 2 y = 1 }
						infantry = { x = 2 y = 2 }
					}
				}
				capital_scope = {
					create_unit = {
						division = "name = \"1. Kurzemes Divizija\" division_template = \"Kajnieku Divizija\" start_experience_factor = 1.0 start_equipment_factor = 1.0"
						owner = PREV
					}
					create_unit = {
						division = "name = \"2. Vidzemes Divizija\" division_template = \"Kajnieku Divizija\" start_experience_factor = 1.0 start_equipment_factor = 1.0"
						owner = PREV
					}
					create_unit = {
						division = "name = \"3. Zemgales Divizija\" division_template = \"Kajnieku Divizija\" start_experience_factor = 1.0 start_equipment_factor = 1.0"
						owner = PREV
					}
					create_unit = {
						division = "name = \"4. Latgales Divizija\" division_template = \"Kajnieku Divizija\" start_experience_factor = 1.0 start_equipment_factor = 1.0"
						owner = PREV
					}
				}
			}
			every_state = {
				limit = {
					OR = {
						is_core_of = UBD
					}
				}
				remove_core_of = UBD
			}
		}
	}
	option = {
		name = ww1_estonia.1.a
		ai_chance = { factor = 100 }
	}
}
