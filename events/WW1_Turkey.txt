			}

			}

				name = "Naci Eldeniz"

				name = "Celâl Bayar"
				desc = "POLITICS_CELAL_BAYAR_DESC"
				picture = "gfx/leaders/TUR/TUR_celal_bayar.dds" 
				expire = "1935.1.1"
				ideology = conservatism
				traits = { trait_conservatism
				}
			}
		}
		set_cosmetic_tag = TUR_REP
		retire_country_leader = yes

			name = "Mustafa Kemal"
			picture = "Portrait_Turkey_Mustafa_Kemal_Ataturk_Military.dds" 
			expire = "1935.1.1" 
			ideology = authoritarian_democracy

		if = {
			limit = { is_puppet = yes }
			OVERLORD = { end_puppet = TUR }
		}
		if = {
			limit = {
				GRE = {
					has_country_flag = great_war_winner
					NOT = { has_war_with = TUR }
					any_owned_state = {
						region = 129
						is_core_of = TUR
					}
				}
			}
			country_event = { id = ww1_greece.6 days = 14 }
		}
		ai_chance = { factor = 100 }
	}
}

# Greco-Turkish War - Turkish Defeat
#######
country_event = {
	id = ww1_turkey.3
	title = ww1_turkey.3.t
	desc = ww1_turkey.3.d
	picture = GFX_report_event_generic_military_parade

	trigger = {
		tag = TUR
		has_defensive_war_with = GRE
		has_global_flag = interwar_greco_turkish_war
		OR = {
			casualties > 100000
			surrender_progress > 0.20
		}
		is_in_faction = no
		NOT = { has_global_flag = interwar_greco_turkish_war_ended }
	}
	fire_only_once = yes
	mean_time_to_happen = { days = 0 }
	
	option = {
		name = ww1_turkey.3.a	#Ask for peace
		GRE = { country_event = ww1_turkey.4 }
		ai_chance = { factor = 10 }
	}
	option = {
		name = ww1_turkey.3.b	#Victory or Death
		add_war_support = 0.10
		ai_chance = { factor = 0 }
	}
}

country_event = {
	id = ww1_turkey.4
	title = ww1_turkey.4.t
	desc = ww1_turkey.4.d
	picture = GFX_report_event_generic_sign_treaty1

	is_triggered_only = yes

	option = {
	    name = ww1_turkey.4.a
		set_global_flag = interwar_greco_turkish_war_ended
		hidden_effect = {
			TUR = { 
				add_war_support = -0.10
				add_stability = -0.10
				retire_country_leader = yes
				create_country_leader = {
					name = "Mustafa İsmet"
					picture = "gfx/leaders/TUR/TUR_ismet_inonu.dds"
					expire = "1935.1.1" 
					ideology = authoritarian_democracy
					traits = { } 
				}
			}
		}
		every_country = {
			limit = {
				OR = {
					original_tag = GRE
					has_war_together_with = GRE
				}
				has_war_with = TUR
			}
			white_peace = TUR
		}
		every_state = {
			limit = { 
				state = 341 
				is_owned_by = TUR
			}
			GRE = { transfer_state = PREV }
		}
		set_global_flag = greco_turkish_war_greece_won
	    ai_chance = { factor = 100 }	
	}
	option = {
		name = ww1_turkey.4.b #ignore
		TUR = { country_event = ww1_turkey.5 }
		ai_chance = { factor = 0 }
	}
}

					region = 129
					is_core_of = TUR
				}

			country_event = { id = ww1_turkey.9 hours = 48 random = 20 }
		}

	        factor = 100
	    }
