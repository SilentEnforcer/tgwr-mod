###Ittihadist government resigns
country_event = {
	id = ww1_turkey.1
	title = ww1_turkey.1.t
	desc = ww1_turkey.1.d
	fire_only_once = yes
	trigger = {
		original_tag = TUR
		has_civil_war = no
		OR = {
			has_country_leader = {
				name = "Ismail Enver Paşa"
				ruling_only = yes
			}
			has_country_leader = {
				name = "Enver-Cemal-Talat Triumvirate"
				ruling_only = yes
			}
			has_country_leader = {
				name = "Mustafa-Enver-Cemal Triumvirate"
				ruling_only = yes
			}
			has_country_leader = {
				name = "Mahmud Şevket Paşa"
				ruling_only = yes
			}
		}
		has_country_flag = great_war_loser
	}
	option = {
		name = generic_ok
		hidden_effect = {
			ENG = {
				country_event = {
					id = ww1_turkey.4
					days = 30
				}
			}
		}
		if = {
			limit = {
				has_country_flag = mustafa_enver_cemal
			}
			set_country_flag = had_mustafa_kemal
		}
		if = {
			limit = {
				OR = {
					has_government = fascism
					has_country_flag = talat_enver_cemal
					has_country_flag = mustafa_enver_cemal
				}
			}
			set_country_flag = had_enver
		}
		every_country = {
			limit = {
				has_country_flag = great_war_winner
				is_major = yes
			}
			TUR = {
				give_military_access = PREV
			}
		}
		set_politics = {
			ruling_party = neutrality
			elections_allowed = no
		}
		set_party_name = {
			ideology = democratic
			long_name = TUR_democratic_party_long
			name = TUR_democratic_party
		}
		retire_country_leader = yes
		clr_country_flag = talat_enver_cemal
		clr_country_flag = mustafa_enver_cemal
		hidden_effect = {
			country_event = {
				id = ww1_turkey.2
				days = 2
			}
			country_event = {
				id = ww1_turkey.10
				days = 18
			}
		}
		set_country_flag = had_ittihadist_government
		set_party_name = {
			ideology = neutrality
			long_name = TUR_ottoman_dynasty_party
			name = TUR_ottoman_dynasty_party
		}
		if = {
			limit = {
				NOT = {
					has_country_flag = TUR_mehmed_v_dies
				}
			}
			create_country_leader = {
				name = "Mehmed V"
				picture = "gfx/leaders/TUR/TUR_mehmed_v.dds"
				expire = "1935.1.1"
				ideology = autocracy
				traits = {
					mehmed_v
					trait_autocracy
				}
			}
		}
		if = {
			limit = {
				has_country_flag = TUR_mehmed_v_dies
			}
			create_country_leader = {
				name = "Mehmed VI"
				picture = "gfx/leaders/TUR/TUR_mehmed_vi.dds"
				expire = "1935.1.1"
				ideology = autocracy
				traits = {
					mehmed_vi
					trait_autocracy
				}
			}
		}
	}
}
