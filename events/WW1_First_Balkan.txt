					}

					days = 1
					id = ww1_firstbalkan.100
				}
			}
			if = {
				limit = {
					SER = {
						has_completed_focus = SER_great_serbia
					}
				}
				news_event = {
					days = 1
					id = ww1_firstbalkan.102
				}
			}
			BUL = {
				add_opinion_modifier = {
					target = TUR
					modifier = small_decrease
				}
			}
			GRE = {
				add_opinion_modifier = {
					target = TUR
					modifier = small_decrease
				}
			}
			MNT = {
				add_opinion_modifier = {
					target = TUR
					modifier = small_decrease
				}
			}
			add_opinion_modifier = {
				target = TUR
				modifier = large_decrease
			}
			TUR = {
				add_opinion_modifier = {
					target = SER
					modifier = large_decrease
				}
				if = {
					limit = {
						has_completed_focus = TUR_coup_of_savior_officers
						NOT = {
							has_completed_focus = TUR_raid_on_the_sublime_porte
							has_completed_focus = TUR_protect_democracy
							has_completed_focus = TUR_ittihadi_muhammedi
							has_completed_focus = TUR_istirak_mukavemet
						}
					}
					add_timed_idea = {
						idea = TUR_idea_chaos_balkans
						days = 20
					}
				}
				add_opinion_modifier = {
					target = GRE
					modifier = small_decrease
				}
				add_opinion_modifier = {
					target = MNT
					modifier = small_decrease
				}
				add_opinion_modifier = {
					target = BUL
					modifier = large_decrease
				}
			}
		}
		# Bulgaria
		if = {
			limit = {
				BUL = {
					has_country_flag = join_balkan_league
				}
			}
			BUL = {
				add_to_war = {
					targeted_alliance = SER
					enemy = TUR
				}
				if = {
					limit = {
						TUR = {
							has_civil_war = yes
						}
					}
					add_to_war = {
						targeted_alliance = SER
						enemy = OTM
					}
				}
			}
		}
		# Greece
		if = {
			limit = {
				GRE = {
					has_country_flag = join_balkan_league
				}
			}
			GRE = {
				add_to_war = {
					targeted_alliance = SER
					enemy = TUR
				}
				if = {
					limit = {
						TUR = {
							has_civil_war = yes
						}
					}
					add_to_war = {
						targeted_alliance = SER
						enemy = OTM
					}
				}
			}
		}
		# Montenegro
		if = {
			limit = {
				MNT = {
					has_country_flag = join_balkan_league
				}
			}
			MNT = {
				add_to_war = {
					targeted_alliance = SER
					enemy = TUR
				}
			}
			if = {
				limit = {
					TUR = {

				}
				add_to_war = {

	}
