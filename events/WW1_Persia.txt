	id = ww1_persia.5
	title = ww1_persia.5.t
	desc = ww1_persia.5.d
	picture = GFX_report_event_american_hunger_march
	is_triggered_only = yes
	option = {
		name = ww1_persia.5.a
		trigger = {
		    NOT = {
			    has_government = fascism
			}
		}	
		start_civil_war = {
			ideology = fascism
			size = 0.6
			capital = 411
			states = all
			states_filter = {
				NOT = {
					state = 266
					state = 416
					state = 418
				}
			}
			keep_political_leader = yes
			hidden_effect = {
				create_country_leader = {
					name = "Persian Central Government"
					picture = "gfx/leaders/PER/PER_gendarmerie.dds"
					expire = "1935.1.1"
					ideology = stratocracy
					traits = {
						eng_civil_war_trait
						trait_stratocracy
					}
				}
				set_popularities = {
					democratic = 35
					neutrality = 10
					fascism = 55
					communism = 0
				}
				set_cosmetic_tag = PER_gendarmerie
				set_country_flag = PER_gendarmerie_flag
				every_other_country = {
					add_ai_strategy = {
						type = protect
						id = PREV
						value = -200
					}
					add_ai_strategy = {
						type = alliance
						id = PREV
						value = -200
					}
				}
				declare_war_on = {
					target = ENG
					type = take_core_state
				}
				declare_war_on = {
					target = RUS
					type = take_core_state
				}
			}
		}
		set_global_flag = persian_gendarmerie_revolt
		ai_chance = {
			factor = 100
		}
	}
	option = {
		name = ww1_persia.5.b
		declare_war_on = {
			target = ENG
			type = take_core_state
		}
		declare_war_on = {
			target = RUS
			type = take_core_state
		}
		add_popularity = {
			ideology = fascism
			popularity = 0.1
		}
		add_war_support = 0.1
		ai_chance = {
			base = 10
			modifier = {
				is_historical_focus_on = yes
				factor = 0
			}
		}
	}
}

# 1921 Persian coup d'état
################################
country_event = {
	id = ww1_persia.6
	title = ww1_persia.6.t
	desc = ww1_persia.6.d
	trigger = {
		tag = PER
		date > 1921.2.1
		OR = {
			has_government = neutrality
			has_government = fascism
		}
		OR = {
			is_subject_of = ENG
			any_state = {
				is_core_of = PER
				is_owned_by = ENG
			}
		}
	}
	mean_time_to_happen = {
		days = 21
	}
	option = {
		name = ww1_persia.6.a
		if = {
			limit = {
				is_subject = yes
			}
			OVERLORD = {
				end_puppet = PREV
			}
		}
		if = {
			limit = {
				has_idea = idea_victors_of_wwi
			}
			remove_ideas = idea_victors_of_wwi
		}
		retire_ideology_leader = neutrality
		set_politics = {
			ruling_party = neutrality
		}
		create_country_leader = {
			name = "Reza Khan"
			picture = "Portrait_Iran_Reza_Shah_Pahlavi.dds"
			expire = "1965.1.1"
			ideology = despotism
			traits = {
				trait_despotism
			}
		}
		every_state = {
			limit = {
				is_core_of = PER
				OR = {
					is_owned_by = ENG
					is_owned_by = RUS
				}
			}
			PER = {
				transfer_state = PREV
			}
		}
		add_political_power = -100
		add_stability = -0.1
		add_war_support = 0.1
		ai_chance = {
			factor = 100
		}
	}
	option = {
		name = ww1_persia.6.b
		if = {
			limit = {
				is_subject_of = ENG
			}
			ENG = {
				end_puppet = PER
			}
		}
		every_state = {
			limit = {
				is_core_of = PER
				OR = {
					is_owned_by = ENG
					is_owned_by = RUS
				}
			}
			PER = {
				transfer_state = PREV
			}
		}
		start_civil_war = {
			ideology = fascism
			size = 0.2
			keep_political_leader = yes
			hidden_effect = {
				create_country_leader = {
					name = "Reza Khan"
					picture = "Portrait_Iran_Reza_Shah_Pahlavi.dds"
					expire = "1965.1.1"
					ideology = stratocracy
					traits = {
						trait_stratocracy
					}
				}
			}
		}
		add_political_power = -200
		add_stability = -0.2
		add_war_support = -0.1
		ai_chance = {
			factor = 0
		}
	}
}

# Imperial State of Persia (Iran from 1935)
#################################
country_event = {
	id = ww1_persia.7
	title = ww1_persia.7.t
	desc = ww1_persia.7.d
	trigger = {
		original_tag = PER
		has_war = no
		OR = {
			is_subject = no
			has_government = neutrality
		}		
		date > 1925.12.1
		has_country_leader = {
			name = "Reza Khan"
			ruling_only = yes
		}
	}
	mean_time_to_happen = {
		days = 25
	}
	option = {
		name = ww1_persia.7.a
		hidden_effect = {
			retire_country_leader = yes
			set_popularities = {
				democratic = 25
				neutrality = 55
				fascism = 20
				communism = 0
			}
			set_politics = {
				ruling_party = neutrality
			}
			set_cosmetic_tag = PER_pahlavi
		}
		set_party_name = {
			ideology = neutrality
			long_name = PER_pahlavi_dynasty
			name = PER_pahlavi_dynasty
		}
		create_country_leader = {
			name = "Reza Shah Pahlavi"
			picture = "Portrait_Iran_Reza_Shah_Pahlavi.dds"
			expire = "1965.1.1"
			ideology = autocracy
			traits = {
				trait_autocracy
			}
		}
	}
}
