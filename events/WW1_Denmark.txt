			remove_ideas = DEN_frederick_viii

		}	
	}
}

# JUTLAND REFERENDUM

country_event = {
	id = ww1_denmark.3
	title = ww1_denmark.3.t
	desc = ww1_denmark.3.d
	picture = GFX_report_event_generic_peaceful_annexation
	fire_only_once = yes
	mean_time_to_happen = {
		days = 90
	}
	trigger = {
		original_tag = DEN
		any_other_country = {
			owns_state = 751
			has_country_flag = great_war_loser
			is_neighbor_of = PREV
			NOT = {
				has_war_with = PREV
				has_subject = PREV
			}
		}
	}
	option = {
		name = ww1_denmark.3.a
		751 = {
			remove_core_of = GER
			remove_core_of = SHL
			transfer_state_to = PREV
		}
		if = {
			limit = {
				has_country_flag = great_war_winner
			}
			58 = {
				transfer_state_to = PREV
			}
		}
		ai_chance = {
			factor = 100
		}
	}
}

# Kingdom of Iceland
country_event = {
	id = ww1_denmark.4
	title = ww1_denmark.4.t
	desc = ww1_denmark.4.d
	picture = GFX_report_event_generic_peaceful_annexation
	fire_only_once = yes
	trigger = {
		original_tag = DEN
		has_war = no
		is_subject = no
		any_owned_state = {
			is_core_of = ICE
		}
		NOT = {
			country_exists = ICE
		}
		has_global_flag = interbellum
	}
	mean_time_to_happen = { days = 60 }
	option = {
		name = ww1_denmark.4.a
		release_autonomy = {
			target = ICE
			autonomy_state = autonomy_dominion
		}
		hidden_effect = {
			ICE = {
				copy_government_of_ROOT = yes
				copy_laws_of_PREV = yes
				inherit_technology = ROOT
				add_ideas = DEN_christian_x
				promote_character = ICE_christian_x
			}
		}
	}
	option = {
		name = ww1_denmark.4.b
		add_political_power = -75
		add_stability = -0.10
	}
}
