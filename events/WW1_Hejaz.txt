					}

							limit = {

							}

							}

								add_state_claim = PREV

						}
					}

				limit = {

						exists = no
					}

						is_core_of = YEM
					}
				}
				YEM = {
					load_oob = "YEM_1910"
					set_country_flag = arab_revolt_member
				}
				if = {
					limit = {
						owns_state = 293
					}
					YEM = {
						transfer_state = 293
					}
					add_state_claim = 293
					else = {
						random_owned_state = {
							limit = {
								is_core_of = YEM
							}
							YEM = {
								transfer_state = PREV
							}
							ROOT = {
								add_state_claim = PREV
							}
						}
					}
				}
			}
		}
	}
	option = {
		name = ww1_hejaz.7.a
		declare_war_on = {
			target = HJZ
			type = annex_everything
		}
		if = {
			limit = {
				ASR = {
					has_country_flag = arab_revolt_member
				}
			}
			declare_war_on = {
				target = ASR
				type = annex_everything
			}
		}
		if = {
			limit = {
				YEM = {
					has_country_flag = arab_revolt_member
				}
			}
			declare_war_on = {
				target = YEM
				type = annex_everything
			}
		}
		news_event = ww1_hejaz.8
	}
	option = {
		name = ww1_hejaz.7.b
		declare_war_on = {
			target = HJZ
			type = annex_everything
		}
		if = {
			limit = {
				ASR = {
					has_country_flag = arab_revolt_member
				}
			}
			declare_war_on = {
				target = ASR
				type = annex_everything
			}
		}
		if = {
			limit = {
				YEM = {
					has_country_flag = arab_revolt_member
				}
			}
			declare_war_on = {
				target = YEM
				type = annex_everything
			}
		}
		news_event = ww1_hejaz.8
		HJZ = { change_tag_from = ROOT }
	}
}

country_event = {
	id = ww1_hejaz.11
	title = ww1_hejaz.11.t
	desc = ww1_hejaz.11.d
	is_triggered_only = yes
	fire_only_once = yes
	option = {
		name = ww1_hejaz.11.a
		FROM = {
			if = {
				limit = {
					OR = {
						has_dlc = "Together for Victory"
						has_dlc = "Waking the Tiger"
					}
				}
				PREV = { set_country_flag = protectorate }
				set_autonomy = {
					target = PREV
					autonomous_state = autonomy_tgwr_informal_protectorate
				}
				else = {
					puppet = SAU
					PREV = {
						add_ideas = protectorate
					}
				}
			}
		}
		add_political_power = 200
		every_other_country = {
			limit = {
				is_neighbor_of = FROM
				is_subject_of = FROM
			}
			diplomatic_relation = {
				country = PREV
				relation = non_aggression_pact
				active = yes
			}
		}
		hidden_effect = {
			set_politics = {
				ruling_party = neutrality
				elections_allowed = no
			}
			set_popularities = {
				neutrality = 100
			}
		}
		ai_chance = {
			base = 100
		}
	}
	option = {
		name = ww1_hejaz.11.b
		FROM = {
			PREV = {
				every_owned_state = {
					limit = {
						is_core_of = PREV
					}
					add_claim_by = PREV.PREV
				}
			}
		}
		ai_chance = {
			base = 0
		}
	}
}
