country_event = {
	id = ww1_generic.2
	title = ww1_generic.2.t
	desc = ww1_generic.2.d
	picture = GFX_report_event_worker_protests

	trigger = {
		has_offensive_war = yes
		casualties > 2500000
		NOT = { has_government = communism }
		NOT = { has_government = fascism }
		NOT = { has_country_flag = high_casualties }
	}
	immediate = {
		set_country_flag = high_casualties
	}
	mean_time_to_happen = {
		days = 720
		modifier = {
			factor = 0.5
			has_manpower < 10000
		}	
		modifier = {
			factor = 0.5
			has_war_support < 0.5
		}
		modifier = {
			factor = 1.5
			has_war_support > 0.75
		}
		modifier = {
			factor = 1.5
			has_stability > 0.5
		}
	}

	option = {
		name = ww1_generic.2.a
		add_war_support = -0.05
		# фашисты и коммунисты не не боятся пацифистов
		if = {
			limit = { has_government = neutrality }
			add_popularity = {
				ideology = communism
				popularity = 0.02
			}
		}
		if = {
			limit = { has_government = democratic }
			add_popularity = {
				ideology = communism
				popularity = 0.02
			}
		}
		clr_country_flag = high_casualties
		ai_chance = { factor = 100 }

country_event = {
	id = ww1_generic.3
	title = ww1_generic.3.t
	desc = ww1_generic.3.d
	picture = GFX_report_event_worker_protests
	trigger = {
		has_war = no	
		num_of_factories > 10
		NOT = { has_government = fascism }
		NOT = { has_country_flag = crisis_in_progress }
		NOT = { has_country_flag = workers_protest }
		if = {
			limit = { has_government = democratic }
			democratic < 0.5
		}
		if = {
			limit = { has_government = neutrality }
			neutrality < 0.5
		}
		if = {
			limit = { has_government = communism }
			communism < 0.5
		}
	}
	immediate = {
		set_country_flag = workers_protest
	}
	mean_time_to_happen = {
		days = 720
		modifier = {
			factor = 4
			has_country_flag = labor_reform
		}
		modifier = {
			factor = 1.5
			has_stability > 0.35
		}
		modifier = {
			factor = 1.5
			has_idea = civilian_economy
		}
		modifier = {
			factor = 0.75
			has_idea = partial_economic_mobilisation
		}
		modifier = {
			factor = 0.75
			has_idea = war_economy
		}
		modifier = {
			factor = 2
			has_government = communism
		}
		modifier = {
			factor = 0.75
			NOT = { has_government = communism }
			communism < 0.1
		}
	}
	option = {
		name = ww1_generic.3.a
		if = {
			limit = { has_government = neutrality }
			add_popularity = {
				ideology = communism
				popularity = 0.02
			}
		}
		if = {
			limit = { has_government = democratic }
			add_popularity = {
				ideology = communism
				popularity = 0.02
			}
		}
		if = {
			limit = { has_government = communism }
			add_popularity = {
				ideology = democratic
				popularity = 0.02
			}
		}
		clr_country_flag = workers_protest
		ai_chance = { factor = 50 }
	}
	option = {
		name = ww1_generic.3.b		
		random_list = {
			50 = { 
				modifier = {
					factor = 0.75
					has_war_support < 0.5
				}
				country_event = ww1_generic.4 #забастовка разогнана
			}
			50 = { 
				country_event = ww1_generic.5 #кровавое воскресенье
			}
		}
		clr_country_flag = workers_protest
		ai_chance = {
			factor = 10
			modifier = {
				factor = 10
				has_government = neutrality
			}
			modifier = {
				factor = 10
				has_government = communism
			}
		}
	}
	option = {
		name = ww1_generic.3.c
		if = {
			limit = { has_country_flag = labor_reform }
			add_political_power = -75
			add_stability = 0.02
		}
		if = {
			limit = { NOT = { has_country_flag = labor_reform } }
			add_political_power = -100
			set_country_flag = labor_reform
		}
		#трудовая реформа снизит риск забастовок
		clr_country_flag = workers_protest
		ai_chance = {
			factor = 10
			modifier = {
				factor = 10
				has_government = democratic
			}
			modifier = {
				factor = 10
				has_political_power > 100
			}
		}
	}
}
