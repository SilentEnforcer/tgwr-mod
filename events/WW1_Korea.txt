# Korea's Declaration of Independence if Japan loses the Great War
country_event = {
	id = ww1_korea.1
	title = ww1_korea.1.t
	desc = ww1_korea.1.d
	fire_only_once = yes
	trigger = {
		original_tag = JAP
		has_country_flag = great_war_loser
		OR = {
			KOR = {
				is_subject_of = JAP
			}
			NOT = {
				country_exists = KOR
			}
		}
		OR = {
			ENG = {
				has_country_flag = approved_korean_independence
			}
			FRA = {
				has_country_flag = approved_korean_independence
			}
			RUS = {
				has_war = no
				owns_state = 408
				has_country_flag = great_war_winner
			}
		}
	}
	mean_time_to_happen = {
		days = 40
	}
	immediate = {
		hidden_effect = {
			release = KOR
			KOR = {
				set_country_flag = korean_independence
				set_politics = {
					ruling_party = democratic
					elections_allowed = no
				}
			}
			if = {
				limit = {
					RUS = {
						has_war = no
						owns_state = 408
						has_country_flag = great_war_winner
					}
				}
				RUS = {
					if = {
						limit = {
							OR = {
								has_dlc = "Together for Victory"
								has_dlc = "Waking the Tiger"
							}
						}
						set_autonomy = {
							target = KOR
							autonomous_state = autonomy_tgwr_informal_protectorate
						}
						KOR = {
							set_country_flag = protectorate
						}
					}
					else = {
						puppet = KOR
						KOR = {
							add_ideas = protectorate
						}
					}
					if = {
						limit = {
							JAP = {
								owns_state = 537
							}
						}
						add_state_core = 537
						transfer_state = 537
					}
					if = {
						limit = {
							JAP = {
								owns_state = 555
							}
						}
						add_state_core = 555
						transfer_state = 555
					}
				}
			}
		}
	}
	option = {
		name = ww1_korea.1.a
		ai_chance = {
			factor = 100
		}
	}
}
