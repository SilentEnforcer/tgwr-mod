						target = PREV
						size = 0.7
						stockpile_ratio = 0.7
						army_ratio = 0.7
						navy_ratio = 0.4
						air_ratio = 0.5
					}
					every_owned_state = {
						limit = {
							is_core_of = PREV.PREV
							NOT = {
								state = 50
							}
						}

					}
				}
			}
		}
		every_other_country = {
			limit = {
				original_tag = GER
				has_country_flag = kapp_putsch
			}
			PREV = {
				declare_war_on = {
					target = PREV
					type = civil_war
				}
			}
		}

			base = 0
			modifier = {
				add = 100

			}

country_event = {
	id = ww1_germany.112
	title = ww1_germany.112.t
	desc = ww1_germany.112.d
	picture = GFX_report_event_ger_civil_war
	fire_only_once = yes
	trigger = {
		original_tag = GER
		has_government = neutrality
		has_country_flag = kapp_putsch
		has_war = no
		any_other_country = {
			original_tag = GER
			has_government = democratic
			exists = no
		}
	}
	mean_time_to_happen = {
		days = 0
	}
	option = {
		name = ww1_germany.112.a
		add_ideas = GER_revive_the_kaiserreich
		custom_effect_tooltip = GER_revive_the_kaiserreich_tt
		hidden_effect = {
			drop_cosmetic_tag = yes
			remove_ideas = GER_idea_freikorps
			if = {
				limit = {
					OR = {
						HOL = {
							is_subject_of = PREV
							any_state = {
								is_core_of = PREV
								PREV.PREV = {
									owns_state = PREV
								}
							}
						}						
					}
				}
				create_country_leader = {
					name = "Wilhelm II"
					desc = "GER_wilhelm_ii_desc"
					picture = "gfx/leaders/GER/GER_wilhelm_ii.dds"
					expire = "1935.1.1"
					ideology = autocracy
					traits = {
						anti_democrat imperial_connections trait_autocracy
					}
				}
				set_party_name = {
					ideology = neutrality
					long_name = GER_neutrality_party
					name = GER_neutrality_party
				}
				else = {
					country_event = wtt_germany.3
				}
			}
		}
		add_named_threat = {
			threat = 5
			name = GER_revive_the_kaiserreich
		}
		ai_chance = {
			factor = 100
		}
	}
	option = {
		name = ww1_germany.112.b
		set_politics = {
			ruling_party = fascism
			elections_allowed = no
		}
		ai_chance = {
			factor = 0
		}
	}
}

news_event = {
	id = ww1_germany.113
	title = ww1_germany.113.t
	desc = ww1_germany.113.d
	picture = GFX_news_event_ger_civil_war
	major = yes
	trigger = {
		has_global_flag = interbellum
		NOT = {
			has_global_flag = prussian_blue
		}
		FRA = {
			has_country_flag = great_war_winner
			any_owned_state = {
				is_core_of = PREV.PREV
				NOT = {
					is_core_of = PREV
				}
			}
		}
		GER = {
			has_country_flag = great_war_loser
			has_completed_focus = GER_resist_to_occupation
			has_idea = GER_idea_reparations_dawes
			NOT = {
				has_idea = strikes_3
			}
		}
	}
	fire_only_once = yes
	immediate = {
	    hidden_effect = {
			every_state = {
				limit = {
					OR = {
						is_owned_by = FRA
						is_owned_by = ENG
						is_owned_by = BEL
					}			
					is_core_of = ROOT
					NOT = {
						is_core_of = FRA
						is_core_of = ENG
						is_core_of = BEL
					}
				}
				GER = {
 					transfer_state = PREV
 				}
 				set_demilitarized_zone = yes
			}
		}
	}
	option = {
		name = ww1_germany.113.a
		trigger = {
			OR = {
				original_tag = GER
				has_country_flag = great_war_loser
			}
		}
	}
	option = {
		name = ww1_germany.113.b
		trigger = {
			NOT = {
				original_tag = GER
				has_country_flag = great_war_loser
			}
		}
	}
}
