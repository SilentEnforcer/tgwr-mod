# TAKE SUDETENLAND
########################
country_event = {
	id = ww1_czechoslovakia.1
	title = ww1_czechoslovakia.1.t
	desc = ww1_czechoslovakia.1.d
	picture = GFX_report_event_generic_read_write
	mean_time_to_happen = {
		days = 30
	}
	fire_only_once = yes
	trigger = {
		tag = CZE
		is_subject = no
		AUH = {
			has_country_flag = great_war_loser
		}
		OR = {
			AUH = {
				exists = yes
				NOT = {
					has_war_with = CZE
				}
				OR = {
					has_completed_focus = AUH_unitary_monarchy
					has_completed_focus = AUH_reform_transleithania
				}
				owns_state = 69
				owns_state = 74
			}
			AUS = {
				exists = yes
				NOT = {
					has_war_with = CZE
				}
				owns_state = 69
				owns_state = 74
			}
		}
	}
	option = {
		name = ww1_czechoslovakia.1.a
		transfer_state = 69
		transfer_state = 74
		every_state = {
			limit = {
				is_owned_by = CZE
				NOT = {
					state = 69
					state = 74
				}
			}
			remove_core_of = AUH
			remove_core_of = AUS
		}
		AUH = {
			if = {
				limit = {
					exists = yes
					OR = {
						has_completed_focus = AUH_unitary_monarchy
						has_completed_focus = AUH_reform_transleithania
					}
				}
				country_event = ww1_czechoslovakia.3
			}
			else = {
				AUS = {
					country_event = ww1_czechoslovakia.3
				}
			}
		}
		ai_chance = {
			factor = 100
		}
	}
}

country_event = {
	id = ww1_czechoslovakia.2
	title = ww1_czechoslovakia.2.t
	desc = ww1_czechoslovakia.2.d
	picture = GFX_report_event_generic_read_write
	fire_only_once = yes
	mean_time_to_happen = {
		days = 53
	}
	trigger = {
		AUH = {
			has_country_flag = great_war_loser
		}
		tag = CZE
		is_subject = no
		exists = yes
		has_war = no
		HUN = {
			exists = yes
			is_subject = no
			NOT = {
				has_war_with = CZE
			}
			owns_state = 71
			owns_state = 70
		}
	}
	option = {
		name = ww1_czechoslovakia.2.a
		transfer_state = 71
		transfer_state = 70
		every_state = {
			limit = {
				is_owned_by = CZE
			}
			remove_core_of = HUN
		}
		add_ideas = CZE_divided_nation
		hidden_effect = {
			# Fix Carpanthian Ruthenia
			if = {
				limit = {
					any_other_country = {
						OR = {
							original_tag = UKR
							original_tag = WUK
						}
						OR = {
							is_subject = yes
							NOT = {
								has_country_flag = great_war_winner
							}
						}
						owns_state = 73
					}
				}
				HUN = {
					transfer_state = 73
				}
			}
		}
	}
}

country_event = {
	id = ww1_czechoslovakia.3
	title = ww1_czechoslovakia.3.t
	desc = ww1_czechoslovakia.3.d
	picture = GFX_report_event_generic_read_write
	fire_only_once = yes
	is_triggered_only = yes
	option = {
		name = ww1_czechoslovakia.3.a
		ai_chance = {
			factor = 100
		}
		if = {
			limit = {
				has_cosmetic_tag = AUS_AUH
			}
			drop_cosmetic_tag = yes
		}
		every_state = {
			limit = {
				is_core_of = ROOT
				OR = {
					state = 69
					state = 74
				}
			}
			remove_core_of = ROOT
		}
	}
	option = {
		name = ww1_czechoslovakia.3.b
		add_political_power = -100
		add_stability = -0.1
		create_wargoal = {
			target = CZE
			type = annex_everything
		}
		ai_chance = {
			base = 0
			modifier = {
				add = 10
				tag = AUH
			}
		}
	}
}
