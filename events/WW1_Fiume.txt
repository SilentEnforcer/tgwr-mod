			FROM = { has_country_flag = fiume_withdraw }
		}

					change_state_after_war = no

						state = 736

						state = 513
						num_provinces = 1
						on_win = ww1_fiume.6
						on_lose = ww1_fiume.5

				}
			}
			add_political_power = -200
			add_stability = -0.15
		}
		else_if = {
			limit = {
				FROM = { has_country_flag = fiume_annex }
			}
			every_state = {
				limit = {
					owns_or_subject_of = yes
					OR = {
						state = 103
						state = 1006
					}
				}
				transfer_state_to = FROM
			}
			FROM = {
				if = {
					limit = {
						owns_state = 163
					}
					163 = { transfer_state_to = ROOT }
				}
				start_border_war = {
					change_state_after_war = no
					attacker = {
						state = 736
						num_provinces = 1
						on_win = ww1_fiume.5
						on_lose = ww1_fiume.6
					}
					defender = {
						state = 513
						num_provinces = 1
						on_win = ww1_fiume.6
						on_lose = ww1_fiume.5
					}
				}
			}
			add_political_power = -100
			add_stability = -0.10
		}
		else = {
			every_state = {
				limit = {
					is_owned_by = FROM
					OR = {
						state = 103
						state = 1006
					}
				}
				transfer_state_to = ROOT
			}
			FROM = {
				if = {
					limit = {
						owns_state = 163
					}
					163 = { transfer_state_to = ROOT }
				}
				start_border_war = {
					change_state_after_war = no
					attacker = {
						state = 736
						num_provinces = 1

						on_lose = ww1_fiume.6
					}

						state = 513
						num_provinces = 1
						on_win = ww1_fiume.6
						on_lose = ww1_fiume.5
					}
				}
			}
			add_political_power = 25
		}
		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				FROM = { has_country_flag = fiume_demand }
			}
			modifier = {
				add = -80
				FROM = { has_country_flag = fiume_annex }
			}
		}
	}
	option = {
		# Disagree
		name = ww1_fiume.4.b
		trigger = {
			FROM = {
				NOT = { has_country_flag = fiume_withdraw }
			}

				target = ROOT

			target = FIU
			type = take_claimed_state
		}
		ai_chance = {
			base = 80
			modifier = {
				factor = 0
				FROM = { has_country_flag = fiume_historical }
			}
		}
	}
	option = {
		# They gave up
		name = ww1_fiume.4.c
		trigger = {
			FROM = { has_country_flag = fiume_withdraw }
		}
		FROM = {

	}

	picture = GFX_report_event_generic_peaceful_annexation
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					has_country_flag = fiume_historical
				}
				set_global_flag = free_state_fiume
				FIU = {
					set_politics = {
						ruling_party = neutrality
						elections_allowed = no
					}
					add_popularity = {
						ideology = neutrality
						popularity = 0.45
					}
					every_other_country = {
						limit = {
							is_neighbor_of = PREV
							NOT = { has_war_with = PREV }
						}
						diplomatic_relation = {
							country = PREV
							relation = non_aggression_pact
						}
						give_guarantee = PREV
					}
					drop_cosmetic_tag = yes
					if = {
						limit = { has_global_flag = league_of_nations }
						join_the_league_of_nations = yes
					}
					ITA_alceste_de_ambris = {
						set_nationality = ITA
					}
					ITA_filippo_tommaso_marinetti = {
						set_nationality = ITA
					}
					ITA_maffeo_pantaleoni = {
						set_nationality = ITA
					}
					ITA_gabriele_d_annunzio = {
						remove_any_country_role_from_character = yes
						set_nationality = ITA
					}
					ITA = { give_guarantee = FIU }
				}
				else = {
					ITA = {
						annex_country = {
							target = FIU
						}
					}
				}
			}
		}
	}
	option = {
		name = ww1_fiume.5.a
		give_guarantee = FIU
	}
}
