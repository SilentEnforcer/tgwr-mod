# Rhineland Occupation End?
################################
country_event = {
	id = prussian.32
	title = prussian.32.t
	desc = prussian.32.d
	picture = GFX_report_event_generic_peaceful_annexation
	trigger = {
		has_country_flag = great_war_winner
		has_global_flag = treaty_of_berlin
		OR = {
			owns_state = 51
			owns_state = 343
		}
		NOT = {
			original_tag = FRA
			has_country_flag = keep_rhineland
		}
	}
	fire_only_once = yes
	option = {
		name = prussian.32.a
		every_owned_state = {
			limit = {
				is_core_of = RHN
				NOT = { state = 793 }
			}
			RHN = {
				give_resource_rights = {
					receiver = ROOT
					state = PREV
				}
				transfer_state = PREV
			}
		}
		RHN = { give_military_access = ROOT }
		give_guarantee = RHN
		add_offsite_building = { type = industrial_complex level = 1 }
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				51 = { is_owned_by = ROOT is_core_of = ROOT }
			}
		}
	}
	option = {
		name = prussian.32.b
		set_country_flag = keep_rhineland
		if = {
			limit = {
				any_owned_state = {
					is_core_of = RHN
					NOT = { is_core_of = ROOT }
				}
			}
			add_political_power = -25
		}
		ai_chance = {
			base = 0
			modifier = {
				add = 10
				51 = { is_owned_by = ROOT is_core_of = ROOT }
			}
		}
	}
}
