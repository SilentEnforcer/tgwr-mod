country_event = {
	id = ww1_turkestan.1
	title = ww1_turkestan.1.t
	desc = ww1_turkestan.1.d
	picture = GFX_report_event_generic_military_parade
	is_triggered_only = yes
	trigger = {
		any_owned_state = {
			NOT = {
				state = 584
				state = 587
			}
			is_core_of = TRK
		}
		TRK = {
			exists = no
		}
	}
	immediate = {
		hidden_effect = {
			every_state = {
				limit = {
					NOT = {
						state = 584
						state = 587
					}
					is_core_of = TRK
					OR = {
						is_owned_by = RUS
						is_owned_by = RUS
					}
				}
				TRK = {
					transfer_state = PREV
				}
			}
			TRK = {				
				inherit_technology = RUS
				delete_obsolete_russian_templates = yes
				load_oob = "TRK_1918"
				add_ideas = {
					disarmed_nation
				}
			}
			set_global_flag = basmach_released
		}
	}
	option = {
		name = ww1_turkestan.1.a
		effect_tooltip = {
			release = TRK
		}
	}
	option = {
		name = ww1_turkestan.1.b
		TRK = {
			change_tag_from = ROOT
		}
		trigger = {
			is_ai = no
		}
	}
}

			limit = {
				KHI = { 
				    has_war_with = SOV

# Askhabad Uprising
country_event = {
	id = ww1_turkestan.101
	title = ww1_turkestan.101.t
	desc = ww1_turkestan.101.d
	picture = GFX_report_event_generic_military_parade
	
	trigger = {
		tag = SOV
		has_government = communism
		has_country_flag = czechoslovak_revolt
		584 = {
			owns_or_subject_of = yes
		}
	}

	mean_time_to_happen = {
		days = 20
	}

	immediate = {
		hidden_effect = {
			every_state = {
				limit = {
					OR = {
						state = 584
						state = 587
					}
				}
				add_core_of = TRC
			}
			584 = {
				transfer_state_to = TRC
			}
			TRC = {
				add_ideas = RUS_russian_civil_war
				inherit_technology = SIB
				delete_obsolete_russian_templates = yes
				hidden_effect = { ENG = { country_event = { id = ww1_turkestan.102 days = 10 } } }
				declare_war_on = {
					target = SOV
					type = civil_war
				}
				load_oob = TRC_uprising			
			}
			SOV = {
				send_equipment = {
					equipment = infantry_equipment
					amount = 5000
					target = TRC
				}
			}
		}
	}
	
	fire_only_once = yes
	
	option = {
		name = ww1_turkestan.101.a
		ai_chance = {
			factor = 999
		}
	}
	option = {
		name = ww1_turkestan.101.b
		trigger = {
			is_ai = no
		}
		TRC = { change_tag_from = ROOT }
	}
}
