# SECOND BALKAN WAR
####################
country_event = {
	id = ww1_secondbalkan.1
	title = ww1_secondbalkan.1.t
	desc = ww1_secondbalkan.1.d
	picture = GFX_report_event_military_planning

	fire_only_once = yes
	is_triggered_only = yes
	option = {
		name = ww1_secondbalkan.1.a
		ai_chance = { factor = 100 }
		BUL = {		## attack !
			declare_war_on = {
				target = SER
				type = take_claimed_state
				generator = {
					106 758 731
				}
			}
			set_country_flag = kis_secondbalkanwar_start
		}
		every_country = {	## every ally of the first balkan war will auto join the serbian war
			limit = {
				is_in_faction_with = SER
				has_country_flag = BW1_winner
			}
			add_to_war = {
				targeted_alliance = SER
				enemy = BUL
			}
			set_country_flag = join_antibulgarian_league
		}
		every_state = {	## serbia claim macedonia
			limit = {
			   is_core_of = MAC
			   is_owned_by = BUL
		   }
		   add_claim_by = SER
	   }
	   SER = { ##faction change name
			if = {
				limit = {
					is_faction_leader = yes
				}
				set_faction_name = anti_bulg_league
			}
	   }
	   set_global_flag = kis_secondbalkan_war ## war started for the world
	   news_event = { days = 1 id = ww1_secondbalkan.100 }
		
	}
}
