	desc = ww1_georgia.7.d

	trigger = {
		tag = ENG
		has_guaranteed = GEO

			RUS = {
				is_subject = no
				OR = {
					has_war_with = GEO
					has_wargoal_against = GEO
				}
				NOT = { has_war_with = ROOT }
			}
			SOV = {
				is_subject = no
				OR = {
					has_war_with = GEO
					has_wargoal_against = GEO
				}

			}
		}
		is_subject = no
	}
	fire_only_once = yes

		name = ww1_georgia.7.a
		diplomatic_relation = {
			country = GEO
			relation = guarantee
			active = no
		}

	}
	option = {
		name = ww1_georgia.7.b
		ai_chance = {
			base = 0
			modifier = {
				add = 50
				OR = {
					has_war_with = RUS
					has_war_with = SOV
					has_wargoal_against = RUS
					has_wargoal_against = SOV
				}
			}
		}
	}
