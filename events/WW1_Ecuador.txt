	desc = {
		text = ww1_ecuador.2.d
		trigger = {
			has_country_flag = Emilio_Estrada_chosen
		}
	}
	desc = {
		text = ww1_ecuador.2.d_2
		trigger = {
			has_country_flag = Flavio_Alfaro_chosen
		}
	}
	picture = GFX_report_event_election_vote
	fire_only_once = yes
	mean_time_to_happen = {
		days = 25
	}
	trigger = {
		original_tag = ECU
		OR = {
			has_country_flag = Flavio_Alfaro_chosen
			has_country_flag = Emilio_Estrada_chosen
		}
		has_country_leader = {
			character = ECU_eloy_alfaro
			ruling_only = yes
		}
		date > 1911.6.5
	}
	option = {
		name = ww1_ecuador.2.a
		set_country_flag = Eloy_Alfaro_resigns
		if = {
			limit = {
				has_country_flag = Flavio_Alfaro_chosen
			}
			promote_character = ECU_flavio_alfaro
		}
		if = {
			limit = {
				has_country_flag = Emilio_Estrada_chosen
			}
			promote_character = ECU_emilio_estrada
			add_political_power = -50
		}
		ai_chance = {
			factor = 30
			modifier = {
				is_historical_focus_on = yes
				factor = 0
			}
		}
	}
	# I am the Senate
	option = {
		name = ww1_ecuador.2.b
		trigger = {
			has_country_flag = Emilio_Estrada_chosen
		}
		add_stability = -0.1
		add_popularity = {
			ideology = neutrality
			popularity = 0.15
		}
		set_country_flag = Eloy_Alfaro_keept_the_power
		ai_chance = {
			factor = 70
		}
		hidden_effect = {
			country_event = {
				id = ww1_ecuador.3
				days = 30
				random = 5
			}
		}
	}
}

		trigger = {
			NOT = {
				has_country_flag = Eloy_Alfaro_keept_the_power

		}

	picture = GFX_report_event_hol_cabinet_gerbrandy

		has_country_leader = {

		}
		date > 1911.12.15
	}
	option = {
		name = ww1_ecuador.4.a
		kill_ideology_leader = democratic
		if = {
			limit = {
				has_country_flag = Eloy_Alfaro_keept_the_power
			}

			set_politics = {
				ruling_party = neutrality

				promote_character = ECU_carlos_concha
			}
			else = {
				promote_character = ECU_flavio_alfaro
			}
		}
	}
