	is_triggered_only = yes
	option = {

			if = {
				limit = {
					OR = {
						has_dlc = "Together for Victory"
						has_dlc = "Waking the Tiger"
					}
				}
				set_autonomy = {
					target = UKR
					autonomous_state = autonomy_tgwr_informal_protectorate
				}
				UKR = { set_country_flag = protectorate } 
				else = {
					puppet = UKR	
					UKR = { add_ideas = protectorate } 
				}
			}
		}
		UKR = {
			set_politics = {
				ruling_party = neutrality
				elections_allowed = no
			}
			promote_character = UKR_pavlo_skoropadskyi
			set_country_flag = ukrainian_hetmanate
			hidden_effect = {
				set_cosmetic_tag = UKR_GETMAN
			}
		}
		every_state = {
			limit = {
				is_claimed_by = UKR
				OR = {
					is_owned_by = RUS
					is_owned_by = SOV
					is_owned_by = BLT
					is_owned_by = POL
				}	
			}
			transfer_state_to = UKR
			remove_core_of = RUS
			remove_core_of = SOV			
		}
		RUS = {
			remove_unit_leader = 76234
			RUS_aleksandr_ragoza = {
				promote_leader = yes
				set_nationality = UKR				
			}
		}	
		ai_chance = { factor = 90 }
	}
	option = {
		name = ww1_ukraine.2.b
		GER = { country_event = { id = ww1_ukraine.3 days = 1 } }
		ai_chance = { 
			factor = 10
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			} 
		}
	}
}

#FALL OF HETMANATE
###################
country_event = {
	id = ww1_ukraine.4
	title = ww1_ukraine.4.t
	desc = ww1_ukraine.4.d
	fire_only_once = yes
	trigger = {
		tag = UKR
		has_government = neutrality
		has_global_flag = great_war_ends
		has_cosmetic_tag = UKR_GETMAN
		has_country_flag = ukrainian_hetmanate	
		GER = {
			OR = {
				has_country_flag = great_war_loser
				is_subject = yes
				exists = no
				has_civil_war = yes
				has_country_flag = third_franco_german_war_loser
			}
		}
	}
	mean_time_to_happen = {
		days = 30
	}
	option = {
		name = ww1_ukraine.4.a
		GER = { end_puppet = UKR }
		if = {
			limit = {
				is_in_faction_with = GER
			}
			leave_faction = yes
		}
		set_politics = {
			ruling_party = democratic
			elections_allowed = yes
		}
		set_popularities = {
			democratic = 50
			fascism = 0
			communism = 40
			neutrality = 10
		}
		set_party_name = {
			ideology = democratic
			long_name = UKR_directory_party							
			name = UKR_directory_party
    	}
		promote_character = UKR_volodymyr_vynnychenko
    	drop_cosmetic_tag = yes
    	clr_country_flag = ukrainian_hetmanate
		clr_country_flag = protectorate
    		ai_chance = { 
    			factor = 100
		}
	}
}
