#1916 elections
country_event = {
	id = ww1_argentina.3
	title = ww1_argentina.3.t
	desc = ww1_argentina.3.d
	picture = GFX_report_event_election_vote
	fire_only_once = yes
	is_triggered_only = yes
	trigger = {
		OR = {
			has_government = neutrality
			has_government = democratic
		}
		date > 1916.1.1
		date < 1917.1.1
	}
	option = {
		name = ww1_argentina.3.a
		trigger = {
			democratic > 0.5
		}
		set_politics = {
			ruling_party = democratic
			elections_allowed = yes
		}
		hidden_effect = {
			set_party_name = {
				ideology = neutrality
				long_name = ARG_conservative_party
				name = ARG_conservative_party
			}
		}
		clr_country_flag = dont_trigger_political_event
		ai_chance = {
			factor = 70
		}
	}
	option = {
		name = ww1_argentina.3.b
		hidden_effect = {
			set_politics = {
				ruling_party = democratic
				elections_allowed = yes
			}
			kill_country_leader = yes
		}
		set_party_name = {
			ideology = democratic
			long_name = ARG_conservative_party
			name = ARG_conservative_party
		}
		add_country_leader_role = {
			Character = ARG_victorino_de_la_plaza
			Promote_leader = yes
			Country_leader = {
				ideology=conservatism
				expire="1935.1.1.1"
				traits = {
					trait_conservatism
				}			
			}
		}
		clr_country_flag = dont_trigger_political_event
		ai_chance = {
			factor = 40
			modifier = {
				is_historical_focus_on = yes
				factor = 0
			}
		}
	}
	option = {
		name = ww1_argentina.3.c
		trigger = {
			communism > 0.4
		}
		set_politics = {
			ruling_party = communism
			elections_allowed = yes
		}
		hidden_effect = {
			set_party_name = {
				ideology = neutrality
				long_name = ARG_conservative_party
				name = ARG_conservative_party
			}
		}
		clr_country_flag = dont_trigger_political_event
		ai_chance = {
			factor = 50
			modifier = {
				is_historical_focus_on = yes
				factor = 0
			}
		}
	}
}
