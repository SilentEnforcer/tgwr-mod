			# Warlord Era for Qing Empire
			else = {
				if = {
					limit = {
						QIN = {
							exists = yes
							is_faction_leader = yes
							NOT = {
								has_war_with = SIK
							}
						}
						CHI = {
							exists = no
						}
					}
					QIN = {
						add_to_faction = SIK
					}
				}
			}
		}
		hidden_effect = {
			create_country_leader = {
				name = "Sheng Shicai"
				picture = "gfx/leaders/SIK/Portrait_Sinkiang_Sheng_Shicai.dds"
				expire = "1935.1.1"
				ideology = vanguardism
				traits = {
					trait_vanguardism
				}
			}
			# East Turkestan will be annexed because there is no unique state for them.
			if = {
				limit = {
					KUM = {
						is_ai = yes
					}
				}
				annex_country = {
					target = KUM
					transfer_troops = yes
				}
				every_state = {
					limit = {
						is_core_of = KUM
					}
					remove_core_of = KUM
					add_core_of = SIK
				}
			}
			else = {
				give_guarantee = KUM
				give_military_access = KUM
				KUM = {
					give_military_access = SIK
				}
				every_state = {
					limit = {
						is_core_of = SIK
					}
					add_core_of = KUM
				}
			}
		}
	}
}
