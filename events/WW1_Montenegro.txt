#Serbia asked to annex Montenegro

	id = ww1_montenegro.7
	title = ww1_montenegro.7.t
	desc = ww1_montenegro.7.d
	is_triggered_only = yes
	fire_only_once = yes
	option = {
		#Yay
		name = ww1_montenegro.7.a
		MNT = {
			every_unit_leader = {
				limit = {
					NOT = {
						has_trait = trait_royal
						has_trait = trait_royalist
					}
				}
				hidden_effect = {
					remove_any_country_role_from_character = yes					
					if = {
						limit = {
							is_field_marshal = no
						}
						demote_leader = yes
						remove_unit_leader_trait = recently_promoted
					}
				}	
				set_nationality = SER
			}
			every_owned_state = {
				limit = {
					is_core_of = MNT
				}
				add_core_of = SER
			}
		}
		annex_country = {
			target = MNT
			transfer_troops = yes
		}
		hidden_effect = {
			MNT = {
				set_politics = {
					ruling_party = neutrality
					long_name = MNT_zeleni_party_long
					name = MNT_zeleni_party

				}
				set_party_name = {
					ideology = fascism
					long_name = MNT_zeleni_party_long
					name = MNT_zeleni_party
				}
				promote_character = MNT_krsto_popovic	
				promote_character = MNT_jovan_plamenac			
				promote_character = MNT_novica_radovic
			}	
		}	
		ai_chance = {
			factor = 100
		}
	}
	option = {
		#Nah
		name = ww1_montenegro.7.b
		add_political_power = 50
		ai_chance = {
			factor = 0
		}
	}
}

#Austrian answer to this all
country_event = {
	id = ww1_montenegro.9
	title = ww1_montenegro.9.t
	desc = ww1_montenegro.9.d
	is_triggered_only = yes
	fire_only_once = yes
	option = {
		#Jej
		name = ww1_montenegro.9.a
		MNT = {
			set_cosmetic_tag = MNT_SER
			every_country = {
				limit = {
					has_war_with = MNT
					NOT = {
						has_war_with = AUS
					}
				}
				white_peace = PREV
			}
			set_country_flag = MNT_serbia
			add_opinion_modifier = {
				target = SER
				modifier = faction_traitor
			}
			add_country_leader_trait = unpopular_monarch
		}
		every_state = {
			limit = {
				is_core_of = SER
			}
			add_core_of = MNT
		}
		every_state = {
			limit = {
				is_owned_by = SER
				is_controlled_by = AUS
			}
			MNT = {
				set_state_controller = PREV
			}
		}
		every_state = {
			limit = {
				is_owned_by = MNT
			}
			add_core_of = SER
		}
		set_autonomy = {
			target = MNT
			autonomy_state = autonomy_dominion
			freedom_level = 0.4
		}
		SER = {
			country_event = ww1_montenegro.10
			add_opinion_modifier = {
				target = MNT
				modifier = faction_traitor
			}
		}
		ai_chance = {
			factor = 40
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
	}
	option = {
		#he is still enemy tho
		name = ww1_montenegro.9.b
		ai_chance = {
			factor = 80
		}
	}
}
