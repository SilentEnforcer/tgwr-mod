#Anslutning
country_event = {
	id = ww1_sweden.12
	desc = ww1_sweden.12.d
	title = ww1_sweden.12.t
	is_triggered_only = yes
	fire_only_once = yes
	#picture =
	option = {
		name = ww1_sweden.12.a
		trigger = {
			has_war = no
			is_subject = no
		}
		ai_chance = {
			factor = 80
			modifier = {
				factor = 0
				OR = {
					strength_ratio = {
						tag = SWE
						ratio < 1
					}
					SWE = {
						has_war = yes
					}
					is_in_faction = yes
				}			
			}
		}
		SWE = {
			annex_country = {
			    target = ROOT
			    transfer_troops = yes
			}			
		}
		every_state = {
			limit = {
				is_core_of = ROOT
			}
			add_core_of = SWE
		}
		if = {
			limit = {
				is_ai = no
				SWE = {
					is_ai = yes
				}
			}
			SWE = { change_tag_from = ROOT }
		}
		news_event = {
			id = ww1_sweden.13
		}
	}
	#resist!
	option = {
		name = ww1_sweden.12.b
		ai_chance = {
			factor = 20
			modifier = {
				factor = 0
				strength_ratio = {
					tag = SWE
					ratio < 0.5
				}
				SWE = {
					has_war = no
				}
				is_in_faction = no
			}
		}
		SWE = {
			create_wargoal = {
				type = annex_everything
				target = ROOT
			}
		}
		every_state = {
			limit = {
				is_core_of = ROOT
			}
			add_claim_by = SWE
		}
	}
}
