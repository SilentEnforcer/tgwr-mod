PER_persian_british_sphere_states = {
	icon = gre_faction_management

	allowed = {
		OR = {
			tag = ENG
			tag = PER
		}
	}
	
	visible_when_empty = yes
	
	on_map_area = {
		targets = {
			410
			414
		}	
		target_trigger = {
			FROM = {
				has_dynamic_modifier = { modifier = per_british_sphere }
			}
		}
		name = PER_persian_british_sphere_states
		zoom = 150
	}
	
	priority = 30
	
	visible = {
		any_state = { 
			has_dynamic_modifier = { modifier = per_british_sphere }
		}
		ENG = {
			is_subject = no
			NOT = {
				has_war_with = PER
			}
		}	
	}
}
