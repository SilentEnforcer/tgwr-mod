	}

	enable = {
		original_tag = GER	
		OR = {
			#historical purposes
			has_active_mission = european_crisis_mission
			## someone is preparing war against France in the faction
			any_country = {
				is_in_faction_with = GER
				is_justifying_wargoal_against = FRA
			}
			# French faction is causing trouble
			any_country = {
				NOT = { is_in_faction_with = GER }
				is_in_faction_with = FRA
				has_added_tension_amount > 5
			}
			AND = {
				is_historical_focus_on = yes
				date > 1914.4.4 
				date < 1915.1.1
			}
		}
	}

	abort = {
		OR = {
			## France is safe
			AND = {
				threat < 0.10
				has_non_aggression_pact_with = FRA
			}
			# We won the world war
			has_country_flag = great_war_winner
			## France is our ally
			is_in_faction_with = FRA
			## nothing happened
			date > 1915.1.1
		}
	}
	####
	#	Russia front must still be there
	####
	ai_strategy = {
		type = prepare_for_war
		id = RUS
		value = 1
	}

	#	BELGIUM

	ai_strategy = {
		type = prepare_for_war
		id = BEL
		value = 1
	}

	ai_strategy = {
		type = antagonize
		id = BEL
		value = 200
	}

	ai_strategy = {
		type = front_unit_request
		tag = BEL
		value = 50
	}

	allowed = {

	}

	enable = {
		original_tag = GER
		OR = {	## be sure to stay on planing all long
			#has_war_with = BEL
			has_active_mission = european_crisis_mission
			has_global_flag = kis_franzferdinand_murdered
		}
		#BEL = { has_capitulated = no }
		OR = {
			has_idea = race_to_the_sea
			has_idea = schlieffenmoltke
			
		}
	}

	abort = {
		OR = {
			# We won the world war
			has_country_flag = great_war_winner
			## The war is over with France
			NOT = {
				has_war_with = FRA
				has_war_with = BEL
			}
			## we are losing ground somewhere
			surrender_progress > 0.10
			USA = {
				AND = {
					has_war_with = GER
					casualties > 30000
				}
			}
			# 1 million death is too much to push
			casualties_inflicted_by = {
				opponent = FRA
				thousands > 1000
			}
			# schlieffen failed
			AND = {
				NOT = { has_idea = schlieffenmoltke }
				NOT = { has_idea = race_to_the_sea }
			}
			
			# Belgium surrendered
			# They would immediately stop the offensive after they capitulate so removed this
			# BEL = { has_capitulated = yes }
		}
	}

	ai_strategy = {
		type = front_control
		tag = BEL
		priority = 2000
		ordertype = front
		execution_type = rush
		execute_order = yes
	}
	ai_strategy = {
