scripted_gui = {
	ottoman_millet_system_decision_gui = {
		context_type = decision_category
		window_name = "ottoman_millet_system_gui_window"
		effects = {
			TUR_millet_system_population_button_click = {
				if = {
					limit = { has_country_flag = TUR_millet_system_population_selected }
					clr_country_flag = TUR_millet_system_population_selected
					set_variable = { TUR_millet_system_population_button = 1 }
				}
				else = {
					set_country_flag = TUR_millet_system_population_selected
					set_variable = { TUR_millet_system_population_button = 2 }
				}
			}
			TUR_millet_system_onmap_button_click = {
				if = {
					limit = { has_country_flag = TUR_millet_system_onmap_selected }
					clr_country_flag = TUR_millet_system_onmap_selected
					set_variable = { TUR_millet_system_onmap_button = 1 }
				}
				else = {
					set_country_flag = TUR_millet_system_onmap_selected
					set_variable = { TUR_millet_system_onmap_button = 2 }
				}
			}
			TUR_millet_system_description_button_click = {
				if = {
					limit = { has_country_flag = TUR_millet_system_description_selected }
					clr_country_flag = TUR_millet_system_description_selected
					set_variable = { TUR_millet_system_description_button = 1 }
				}
				else = {
					set_country_flag = TUR_millet_system_description_selected
					set_variable = { TUR_millet_system_description_button = 2 }
				}
			}
		}
		triggers = {
			TUR_millet_system_description_button_click_enabled = { has_country_flag = TUR_millet_system_onmap_selected }
			TUR_millet_system_description_icon_click_enabled = { has_country_flag = TUR_millet_system_onmap_selected }
		}
		properties = {
			TUR_millet_system_population_button = { frame = TUR_millet_system_population_button }
			TUR_millet_system_onmap_button = { frame = TUR_millet_system_onmap_button }
			TUR_millet_system_description_button = { frame = TUR_millet_system_description_button }
		}
	}
}
